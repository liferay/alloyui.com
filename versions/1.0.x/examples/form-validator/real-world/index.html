<form id="myForm">
  <p>
    <label class="aui-field-label" for="name">Name:</label>
    <input type="text" name="name" id="name" />
  </p>
  <p>
    <label class="aui-field-label" for="gender1">Gender:</label>
    <select name="gender" id="gender1">
      <option value="">- Select -</option>
      <option value="male">Male</option>
      <option value="female">Female</option>
    </select>
  </p>
  <p>
    <label class="aui-field-label" for="email">Email:</label>
    <input type="text" name="email" id="email" />
  </p>
  <p>
    <label class="aui-field-label" for="emailConfirmation">Email (confirmation):</label>
    <input type="text" name="emailConfirmation" id="emailConfirmation" />
  </p>
  <p>
    <label class="aui-field-label" for="url">Site URL:</label>
    <input class="aui-field-url" type="text" name="url" id="url" />
  </p>
  <p>
    <input class="aui-button-input" type="submit" value="Submit" />
    <input class="aui-button-input" type="reset" value="Reset" />
  </p>
</form>
AUI().use(
  'aui-form-validator',
  function(A) {
    var rules = {
      email: {
        email: true,
        required: true
      },
      emailConfirmation: {
        email: true,
        equalTo: '#email',
        required: true
      },
      gender: {
        required: true
      },
      name: {
        rangeLength: [2, 50],
        required: true
      },
      url: {
        url: true
      }
    };

    var fieldStrings = {
      email: {
        required: 'Type your email in this field.'
      },
      gender: {
        required: 'The gender is required.'
      },
      name: {
        required: 'Required field with a custom message.'
      }
    };

    new A.FormValidator(
      {
        boundingBox: '#myForm',
        fieldContainer: 'p',
        fieldStrings: fieldStrings,
        rules: rules
      }
    );
  }
);
